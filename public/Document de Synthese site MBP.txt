Section 1 : Contexte du projet
Client / Association : FDD Ma Belle Promo (MBP) — Association des diplômés de la Faculté de Droit du Développement (FDD), Université de Lomé, Togo. Promotion 1994–2000. Fondée le 1er décembre 2018, reconnue officiellement le 03 octobre 2019 (N°0920/MATDCL-SG-DLPAP-DOCA).

Présidente : Fabienne SENAYA-ATAYI

Objectif du site : Remplacer le site Wix existant par un site React moderne, autonome, administrable sans développeur via un dashboard intégré. Le site doit refléter les valeurs de l'association (solidarité, droit, jeunesse, engagement) et permettre la gestion des membres, actualités, événements, projets et médias.

Public cible : Membres de la promotion (Lomé + diaspora), partenaires institutionnels, donateurs potentiels, jeunes juristes togolais.

Section 5 : Décisions importantes prises
localStorage banni du Dashboard — useCrud lit/écrit uniquement sur Supabase. Au montage, chaque clé mbp_store_* est purgée du localStorage. Décision prise après 2 jours de bugs en cascade dus au stockage base64 (5 Mo limite dépassée).

Photos stockées dans Supabase Storage, jamais en base64 — uploadImage() retourne une URL publique. Le champ image d'un article contient toujours une URL, jamais une chaîne base64.

useContent (pages publiques) ≠ useCrud (dashboard) — pages publiques = lecture seule depuis Supabase avec cache localStorage acceptable. Dashboard = Supabase strict.

Contenu article : détecter HTML vs Markdown — article.contenu.trim().startsWith("<") → dangerouslySetInnerHTML, sinon <ReactMarkdown>.

Vidéos : tableau videos[] remplace le champ unique youtube — rétrocompatibilité maintenue (fallback sur article.youtube si videos vide).

Lightbox : index-based, pas URL-based — useState(null) pour l'index courant, useCallback pour prev/next, useEffect pour les touches clavier. Déclaration de galleryPhotos obligatoirement AVANT tous les hooks.

HeroSection original sauvegardé dans HeroSection.original.jsx — peut être restauré à tout moment.

Section 7 : Code clé produit
useCrud — Supabase uniquement, sans localStorage

function useCrud(storeKey, staticData) {
  const KEY = "mbp_store_" + storeKey;
  const [items, setItems] = useState(staticData);
  const [loading, setLoading] = useState(true);
  useEffect(() => {
    try { localStorage.removeItem(KEY); } catch {}
    sbGet(KEY).then(remote => {
      if (remote !== null) setItems(mergeWithStatic(remote, staticData));
      setLoading(false);
    });
  }, [KEY]);
  function save(data) { setItems(data); sbSet(KEY, data); }
  function add(item) { save([...items, { ...item, id: item.id || Date.now() }]); }
  function update(id, data) { save(items.map(it => String(it.id) === String(id) ? { ...it, ...data } : it)); }
  function remove(id) { if (confirm("Supprimer ?")) save(items.filter(it => String(it.id) !== String(id))); }
  return { items, add, update, remove, save };
}
uploadToStorage — Supabase Storage partagé

async function uploadToStorage(file, folder) {
  const ext = file.name.split(".").pop().toLowerCase();
  const filename = `${folder}/${Date.now()}-${Math.random().toString(36).slice(2)}.${ext}`;
  const { error } = await supabase.storage.from("mbp-media")
    .upload(filename, file, { cacheControl: "3600", upsert: false });
  if (error) throw error;
  const { data: { publicUrl } } = supabase.storage.from("mbp-media").getPublicUrl(filename);
  return publicUrl;
}
export async function uploadImage(file) { return uploadToStorage(file, "images"); }
export async function uploadVideo(file) { return uploadToStorage(file, "videos"); }
Parser YouTube robuste

const toEmbed = (url) => {
  try {
    const u = new URL(url);
    if (u.hostname.includes("youtube.com") || u.hostname.includes("youtu.be")) {
      const id = u.searchParams.get("v") || u.pathname.replace(/^\//, "").split("/").pop();
      return id ? `https://www.youtube.com/embed/${id}` : null;
    }
    if (u.hostname.includes("vimeo.com")) {
      const id = u.pathname.replace(/^\//, "").split("/")[0];
      return id ? `https://player.vimeo.com/video/${id}` : null;
    }
  } catch {}
  return null;
};
